<script>
    let chart;
    let history = [72, 68, 85, 90, 78, 88, 0];

    // Inicializa√ß√£o do App
    window.onload = function() {
        initChart();
        updateClock();
        setInterval(updateClock, 1000);
    };

    function initChart() {
        const ctx = document.getElementById('mainChart').getContext('2d');
        chart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: ['S','T','Q','Q','S','S','D'],
                datasets: [{ 
                    data: history, 
                    borderColor: '#8b5cf6', 
                    borderWidth: 3,
                    tension: 0.4,
                    fill: true,
                    backgroundColor: 'rgba(139, 92, 246, 0.1)',
                    pointRadius: 0
                }]
            },
            options: { 
                plugins: { legend: { display: false } },
                scales: { 
                    y: { display: false, min: 0, max: 100 },
                    x: { grid: { display: false }, ticks: { color: 'rgba(255,255,255,0.2)' } }
                }
            }
        });
    }

    // Gerenciamento de Som
    function toggleAudio(type, el) {
        const rain = document.getElementById('sndRain');
        const waves = document.getElementById('sndWaves');
        document.querySelectorAll('.audio-item').forEach(i => i.classList.remove('active'));

        if(type === 'rain') {
            waves.pause();
            if(rain.paused) { rain.play(); el.classList.add('active'); } else rain.pause();
        } else {
            rain.pause();
            if(waves.paused) { waves.play(); el.classList.add('active'); } else waves.pause();
        }
    }

    // Modo de Sono (O que acontece antes de deitar)
    function startSleepMode() {
        document.getElementById('sleepOverlay').style.display = 'flex';
        // Aqui voc√™ pode adicionar um comando para manter a tela ligada se desejar
    }

    function stopSleepMode() {
        document.getElementById('sleepOverlay').style.display = 'none';
        
        // Simular processamento da noite
        const newScore = Math.floor(Math.random() * (98 - 80 + 1)) + 80;
        document.getElementById('score').innerText = newScore;
        
        history.shift();
        history.push(newScore);
        chart.update();
        
        alert("Bom dia! Seu score de sono foi calculado.");
    }

    function updateClock() {
        const now = new Date();
        const time = now.getHours().toString().padStart(2, '0') + ":" + now.getMinutes().toString().padStart(2, '0');
        document.getElementById('liveClock').innerText = time;
    }
</script>
